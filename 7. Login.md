## Service
user.service.ts
```typescript
async login(loginInput:LoginInput){
      const user = await this.userRepo.findOne({where:{email:loginInput.email}})
      if(!user){
        return errorMessage
      }
      const checkPassword = await bcrypt.compare(loginInput.password, user.password)

      if(!checkPassword){
        return errorMessage
      }
    }
```
